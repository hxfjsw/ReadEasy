## 2. 技术栈选型

| 层级 | 技术方案 | 选型理由 |
|------|---------|---------|
| 桌面框架 | **Electron** | 跨平台、成熟生态、React友好 |
| 前端框架 | **React 18** + TypeScript | 组件化开发、类型安全 |
| 构建工具 | **Vite** | 快速冷启动、HMR支持 |
| UI组件库 | **Ant Design** | 丰富的桌面端组件 |
| 状态管理 | **Zustand** | 轻量、TypeScript友好 |
| 样式方案 | **Tailwind CSS** | 原子化CSS、快速开发 |
| 本地存储 | **SQLite** (better-sqlite3) | 轻量、结构化查询 |
| ORM | **Drizzle ORM** | 类型安全、轻量级 |
| 电子书解析 | **epub.js** + 自定义解析器 | 支持EPUB/MOBI/TXT |
| AI服务 | **OpenAI Compatible API** | 支持多种AI服务提供商 |

#### 4.1.2 核心功能实现

```typescript
// electron/services/ai/openai.ts
import OpenAI from 'openai';

export class OpenAIService {
  private client: OpenAI;
  private config: AIConfig;

  constructor(config: AIConfig) {
    this.config = config;
    this.client = new OpenAI({
      apiKey: config.apiKey,
      baseURL: config.baseURL, // 支持自定义端点
    });
  }

  // 单词释义查询
  async getWordDefinition(word: string, context?: string): Promise<WordDefinition> {
    const prompt = this.buildDefinitionPrompt(word, context);
    const response = await this.client.chat.completions.create({
      model: this.config.model,
      messages: [{ role: 'user', content: prompt }],
      temperature: 0.3,
      response_format: { type: 'json_object' },
    });
    return JSON.parse(response.choices[0].message.content);
  }

  // 句子翻译
  async translateSentence(sentence: string): Promise<string> {
    const prompt = `Translate the following English sentence to Chinese:\n${sentence}`;
    const response = await this.client.chat.completions.create({
      model: this.config.model,
      messages: [{ role: 'user', content: prompt }],
      temperature: 0.3,
    });
    return response.choices[0].message.content;
  }

  // 词汇难度分析
  async analyzeVocabularyLevel(text: string): Promise<VocabularyAnalysis> {
    const prompt = this.buildVocabularyAnalysisPrompt(text);
    const response = await this.client.chat.completions.create({
      model: this.config.model,
      messages: [{ role: 'user', content: prompt }],
      temperature: 0.3,
      response_format: { type: 'json_object' },
    });
    return JSON.parse(response.choices[0].message.content);
  }

  // 生成例句
  async generateExampleSentence(word: string, level: string): Promise<string> {
    const prompt = `Generate a simple example sentence using the word "${word}" suitable for ${level} level English learners.`;
    const response = await this.client.chat.completions.create({
      model: this.config.model,
      messages: [{ role: 'user', content: prompt }],
      temperature: 0.7,
    });
    return response.choices[0].message.content;
  }
}
```

```typescript
// src/services/dictionary.ts

interface DictionaryService {
  // 查询单词
  lookup(word: string): Promise<WordDefinition | null>;
  
  // 批量查询
  lookupBatch(words: string[]): Promise<Map<string, WordDefinition>>;
  
  // 播放发音
  playAudio(word: string, accent: 'uk' | 'us'): Promise<void>;
}

// 本地词典实现
export class LocalDictionary implements DictionaryService {
  private db: Database;

  async lookup(word: string): Promise<WordDefinition | null> {
    // 1. 查询本地数据库
    const result = await this.db.query.words.findFirst({
      where: eq(words.word, word.toLowerCase()),
    });

    if (result) {
      return this.formatDefinition(result);
    }

    // 2. 本地未找到，使用AI服务查询
    return null;
  }
}

// 聚合词典服务（本地 + AI）
export class AggregatedDictionary implements DictionaryService {
  private localDict: LocalDictionary;
  private aiService: OpenAIService;

  async lookup(word: string, context?: string): Promise<WordDefinition> {
    // 1. 先查本地
    const localResult = await this.localDict.lookup(word);
    if (localResult) {
      return localResult;
    }

    // 2. 本地未找到，使用AI查询
    const aiResult = await this.aiService.getWordDefinition(word, context);
    
    // 3. 缓存到本地数据库
    await this.cacheWord(aiResult);
    
    return aiResult;
  }
}
```

```typescript
        <Form.Item label="API Key">
          <Input.Password
            value={aiConfig.apiKey}
            onChange={(e) => setAIConfig({ ...aiConfig, apiKey: e.target.value })}
          />
        </Form.Item>

        <Form.Item label="模型">
          <Input
            value={aiConfig.model}
            onChange={(e) => setAIConfig({ ...aiConfig, model: e.target.value })}
            placeholder="gpt-3.5-turbo"
          />
        </Form.Item>

        <Form.Item label="Temperature">
          <Slider
            min={0}
            max={1}
            step={0.1}
            value={aiConfig.temperature}
            onChange={(value) => setAIConfig({ ...aiConfig, temperature: value })}
          />
        </Form.Item>

        <Button onClick={handleTestConnection}>测试连接</Button>
        {testResult && <Alert message={testResult} />}
      </Form>
    </div>
  );
};
```
